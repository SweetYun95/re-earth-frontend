RE-Earth-api/
├─ src/
│  ├─ app.js                   // Express 부트스트랩(라우팅/미들웨어/에러)
│  ├─ server.js                // http + socket.io 서버 구동
│  ├─ config/
│  │  ├─ env.js                // .env 로더
│  │  ├─ db.js                 // Sequelize 연결
│  │  ├─ logger.js             // winston 로거
│  │  └─ swagger.js            // API 문서 설정
│  ├─ auth/
│  │  ├─ passport.js           // JWT, Google, Kakao 전략 설정  ← *OAuth 포함*
│  │  └─ middleware.js         // isAuth, isAdmin, rateLimit 등
│  ├─ models/                  // Sequelize 모델들
│  │  ├─ User.js
│  │  ├─ Item.js               // 포인트 샵 상품
│  │  ├─ Order.js, OrderItem.js
│  │  ├─ Post.js, Qna.js
│  │  ├─ PointLedger.js        // 포인트 원장(적립/사용/기부 기록)
│  │  ├─ Donation.js           // 기부 기록(+리워드)
│  │  ├─ MarketItem.js         // 장터 상품
│  │  ├─ ExchangeRequest.js    // 교환 요청(상태: PENDING/ACCEPTED/REJECTED)
│  │  ├─ ChatRoom.js, ChatMsg.js // 거래 쌍 1:1 채팅
│  │  └─ BinLocation.js        // 의류수거함 좌표(캐시/미러링)
│  ├─ routes/                  // REST 엔드포인트
│  │  ├─ auth.route.js         // 로그인/회원가입/OAuth 콜백
│  │  ├─ users.route.js        // 내정보/관리자 회원관리
│  │  ├─ items.route.js        // 샵 상품 CRUD(관리자)
│  │  ├─ orders.route.js       // 주문 생성/취소(포인트 결제)
│  │  ├─ points.route.js       // 적립/조회/기부/리워드
│  │  ├─ posts.route.js        // 게시글, QnA
│  │  ├─ kpi.route.js          // KPI/기부현황
│  │  ├─ market.route.js       // 장터 CRUD
│  │  └─ exchange.route.js     // 교환 요청/응답
│  ├─ services/                // 비즈니스 로직 계층
│  │  ├─ auth.service.js
│  │  ├─ point.service.js      // 포인트 정합/트랜잭션 처리
│  │  ├─ order.service.js
│  │  ├─ donation.service.js
│  │  ├─ market.service.js
│  │  └─ kpi.service.js
│  ├─ adapters/                // 외부 API 어댑터 (티머니/카카오/공공데이터)
│  │  ├─ tmoney.adapter.js
│  │  ├─ kakao.adapter.js      // OAuth, 지도(좌표 변환/정규화)
│  │  └─ publicdata.adapter.js // 수거함 API → BinLocation 저장/조회
│  ├─ sockets/
│  │  ├─ index.js              // io 초기화, 네임스페이스/미들웨어
│  │  └─ trade.namespace.js    // 거래관련(장터 교환/채팅) 이벤트 정의
│  ├─ middlewares/
│  │  ├─ error.js              // 에러 핸들러
│  │  └─ upload.js             // multer (상품/장터 이미지)
│  └─ utils/
│     ├─ jwt.js                // JWT 발급/검증
│     ├─ geo.js                // GPS 검증/거리계산(Haversine)
│     └─ types.js
├─ .env                        // DB_URL, JWT_SECRET, OAUTH KEYS 등
└─ sequelize.config.js
